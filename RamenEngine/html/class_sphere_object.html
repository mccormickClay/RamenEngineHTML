Detailed Description

A sphere object that will change colors depending on two different alarms.

class SphereObject : public GameObject { public: SphereObject(); virtual ~SphereObject(); virtual void Draw(); void Alarm0(); void Alarm1(); private: GraphicsObject_WireframeConstantColor* GraphicObj; Vect currentColor; Matrix world; Matrix Rot; Matrix Scale; Vect Pos; };

.cpp file

#include "SphereObject.h"
#include "..\src\AssetManager.h"
SphereObject::SphereObject()
{
// Asks Ramen to be able to Register for Drawing
SubmitDrawRegistration();
// Asks Ramen to be able to Register for an Alarm that will
// execute in 3 secs (as a float).
SubmitAlarmRegistration(3, AlarmableManager::ALARM_0);
// Setting the color of the GraphicObject to be easily changed.
currentColor = Color::OrangeRed;
GraphicObj = new GraphicsObject_WireframeConstantColor(AssetManager::GetModel("premadeSphere"), AssetManager::GetShader("colorConstant"), currentColor);
// Setting the SRT of its matrix to world.
Scale.set(SCALE, 15.0f, 15.0f, 15.0f);
Rot.set(IDENTITY);
Pos.set(0, 0, 0);
world = Scale * Rot * Matrix(TRANS, Pos);
// Giving the world matrix to the Graphic Object.
GraphicObj->SetWorld(world);
}
SphereObject::~SphereObject()
{
// Asks Ramen to stop Drawing this object on screen
SubmitDrawDeregistration();
// delete the created GraphicObj in the constructor
delete GraphicObj;
}
void SphereObject::Draw()
{
//CameraTool is accessable by any GameObject to
// easily render and to set the camera.
GraphicObj->Render(CameraTool::GetCurrentCamera());
}
void SphereObject::Alarm0()
{
// For Demo uses
DebugMsg::out("\nSphere is now White");
currentColor = Color::White;
// Sets the color of the GraphicObject
GraphicObj->pWireColor->set(currentColor);
// Setting a new alarm (ALARM_1) to have different
// alarm execute.
Alarmable::SubmitAlarmRegistration(3, AlarmableManager::ALARM_1);
}
void SphereObject::Alarm1()
{
// For Demo uses
DebugMsg::out("\nSphere is now Orange");
currentColor = Color::OrangeRed;
// Sets the color of the GraphicObject
GraphicObj->pWireColor->set(currentColor);
// Setting the old Alarm again. When exicuted the alarm
// Automatically registers itself.
// Submitting for ALARM_O again will make a constant loop.
Alarmable::SubmitAlarmRegistration(3, AlarmableManager::ALARM_0);
}
