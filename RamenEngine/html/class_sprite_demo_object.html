Example

void Bullet::Update()
{
Matrix world;
Vect Trans;
Trans += Vect(0, 0, 1) * Rot * BulletSpeed;
world = Scale* Rot * Matrix(TRANS, Trans);
GraphicsObj->SetWorld(world);
}
