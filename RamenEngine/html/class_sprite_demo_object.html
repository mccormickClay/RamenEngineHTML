Detailed Description

This Demo shows how to ask Ramen to use the Draw2D function inside of a GameObject. It also shows how to draw on that function and the differences between Draw and Draw2D. Draw2D will always be infront of object that were Rendered with Draw.

class SpriteDemoObject : public GameObject
{
public:
SpriteDemoObject();
virtual ~SpriteDemoObject();
void Initialize();
virtual void Update();
virtual void keyPressed(AZUL_KEY k);
virtual void Draw();
virtual void Draw2D();
virtual void Delete();
private:
// 2D Objects
GraphicsObject_Sprite* alien;
Sprite* spr_Alien;
GraphicsObject_SpriteBox* box;
// 3D Objects
Vect currentColor;
GraphicsObject_WireframeConstantColor* GraphicObj;
};

SpriteDemoObject::SpriteDemoObject()
{
Updatable::SubmitUpdateRegistration();
Drawable::SubmitDrawRegistration();
Spritable::Submit2DDrawRegistration();
Inputable::SubmitKeyRegistration(AZUL_KEY::KEY_1, KeyInput::KEY_PRESS);
Inputable::SubmitKeyRegistration(AZUL_KEY::KEY_2, KeyInput::KEY_PRESS);
// Graphics_Sprite
alien = new GraphicsObject_Sprite(AssetTool::Model("spriteModel"), AssetTool::Shader("sprite"), AssetTool::Image("Alien"), new Rect(100.0f, 0.0f, 150.0f, 150.0f));
//Graphics_SpriteBox
Vect color = Color::White;
box = new GraphicsObject_SpriteBox(AssetTool::Model("spriteModel"), AssetTool::Shader("spriteBox"), new Rect(550.0f, 500.0f, 50.0f, 150.0f), color);
//3D GraphicObject
currentColor = Color::OrangeRed;
GraphicObj = new GraphicsObject_WireframeConstantColor(AssetTool::Model("premadeSphere"), AssetTool::Shader("colorConstant"), currentColor);
spr_Alien = SpriteTool::Create(AssetTool::Image("Alien"), 100.0f, 100.0f);
spr_Alien->SetPosition(300.0f, 300.0f);
spr_Alien->SetFactorSize(2.0f, 2.0f);
Matrix Scale;
Matrix Rot;
Vect Pos;
Matrix world;
// Setting the SRT of its matrix to world.
Scale.set(SCALE, 15.0f, 15.0f, 15.0f);
Rot.set(IDENTITY);
Pos.set(0, 0, 0);
world = Scale * Rot * Matrix(TRANS, Pos);
// Giving the world matrix to the Graphic Object.
GraphicObj->SetWorld(world);
}
void SpriteDemoObject::keyPressed(AZUL_KEY k)
{
switch (k)
{
case AZUL_KEY::KEY_1:
SpriteTool::Create(AssetTool::Image("Stitch"), 50.0f, 50.0f);
break;
case AZUL_KEY::KEY_2:
DebugMsg::out("\n<2> pressed", k);
break;
default:
DebugMsg::out("\nERROR: Key %i unprocessed", k);
}
}
void SpriteDemoObject::Initialize()
{
}
void SpriteDemoObject::Update()
{
Matrix world;
Matrix RotZ;
Matrix Scale;
Matrix Trans;
// Alien Graphic_Sprite
// Starting point needs to be the starting point of the rectangle
static float AlienPosX = alien->origPosX;
static float AlienPosY = alien->origPosY;
static float AlienAngle = 0.0f;
AlienAngle += 0.1f;
AlienPosX += 2.0f;
if (AlienPosX > 800.0f)
AlienPosX = 0.0f;
AlienPosY += 1.0f;
if (AlienPosY > 600.0f)
AlienPosY = 0.0f;
Scale.set(SCALE, 1.0f, 1.0f, 1.0f);
RotZ.set(ROT_Z, AlienAngle);
Trans.set(TRANS, AlienPosX, AlienPosY, 0.0f);
world = Scale * RotZ * Trans;
alien->SetWorld(world);
// Sprite Class
spr_Alien->AddRotation(0.1f);
//spr_Alien->SetRotation(90.0f);
spr_Alien->Move(2.0f);
if (spr_Alien->GetXPos() > 800.0f)
spr_Alien->SetPosition(0.0f, spr_Alien->GetYPos());
spr_Alien->Move(0.0f, 1.0f);
if (spr_Alien->GetYPos() > 600.0f)
spr_Alien->SetPosition(spr_Alien->GetXPos(), 0.0f);
// Sprite Box
Scale.set(SCALE, 1.0f, 1.0f, 1.0f);
RotZ.set(ROT_Z, 0.0f);
Trans.set(TRANS, box->origPosX, box->origPosY, 0.0f);
world = Scale * RotZ * Trans;
box->SetWorld(world);
}
void SpriteDemoObject::Draw()
{
GraphicObj->Render(CameraTool::GetCurrentCamera());
}
void SpriteDemoObject::Draw2D()
{
box->Render(CameraTool::GetCurrent2DCamera());
spr_Alien->Render();
alien->Render(CameraTool::GetCurrent2DCamera());
}
void SpriteDemoObject::Delete()
{
Updatable::SubmitUpdateDeregistration();
Drawable::SubmitDrawDeregistration();
Spritable::Submit2DDrawDeregistration();
Inputable::SubmitKeyDeregistration(AZUL_KEY::KEY_1, KeyInput::KEY_PRESS);
Inputable::SubmitKeyDeregistration(AZUL_KEY::KEY_2, KeyInput::KEY_PRESS);
delete alien;
spr_Alien->RemoveAndRecycle();
delete box;
delete GraphicObj;
}
SpriteDemoObject::~SpriteDemoObject()
{
Delete();
}
